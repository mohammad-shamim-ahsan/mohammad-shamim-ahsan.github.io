<!-- JS: attaches event handlers; safe for Jekyll (no inline onclick) -->
<script>
document.addEventListener('DOMContentLoaded', function () {
  // open modal when any .bib-btn is clicked
  document.body.addEventListener('click', function (e) {
    const btn = e.target.closest && e.target.closest('.bib-btn');
    if (!btn) return;
    const key = btn.dataset.bib;
    if (!key) return;
    // get the bib JSON from the hidden script tag
    const script = document.getElementById('bib-' + key);
    if (!script) {
      alert('BibTeX not found: ' + key);
      return;
    }
    try {
      const data = JSON.parse(script.textContent);
      const text = data.bib || '';
      document.getElementById('bib-pre').textContent = text;
      document.getElementById('bib-modal').style.display = 'block';
    } catch (err) {
      console.error(err);
      alert('Error reading BibTeX');
    }
  });

  // close modal controls
  function closeModal() {
    document.getElementById('bib-modal').style.display = 'none';
  }
  document.getElementById('bib-close').addEventListener('click', closeModal);
  document.getElementById('bib-close-2').addEventListener('click', closeModal);
  // click outside modal content to close
  document.getElementById('bib-modal').addEventListener('click', function (ev) {
    if (ev.target === this) closeModal();
  });

  // copy to clipboard
  document.getElementById('bib-copy').addEventListener('click', function () {
    const text = document.getElementById('bib-pre').textContent;
    if (!navigator.clipboard) {
      // fallback
      const ta = document.createElement('textarea');
      ta.value = text;
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
      alert('Copied to clipboard');
      return;
    }
    navigator.clipboard.writeText(text).then(function () {
      alert('Copied to clipboard');
    }, function () {
      alert('Copy failed');
    });
  });
});
</script>
